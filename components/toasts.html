HTML

JavaScript 

var tags = riot.mount('rg-toasts', {
  toasts: {
    position: 'bottomright|bottomleft|topleft|topright',
    toasts: [{
      type: 'error',
      text: 'Made you look!',
      sticky: true // Turn off timeout
    }, {
      type: 'warning',
      text: 'Careful now...'
    }, {
      type: 'success',
      text: 'You did it!'
    }, {
      type: 'info',
      text: 'Oops!',
      timeout: 4000 // Default to 6000 if not set
    }]
  }
})

tags[0].on('close', function (toast) { ... })
       .on('select', function (toast) { ... })
